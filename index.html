// Función principal para recibir peticiones desde tu página en GitHub function doPost(e) { let respuesta = { exito: false, mensaje: "Acción no reconocida" }; try { // Parseamos los datos que envía la página web let parametros = JSON.parse(e.postData.contents); let accion = parametros.accion; if (accion === "login") { respuesta = validarUsuario(parametros.usuario, parametros.password); } else if (accion === "obtenerEstudiantes") { respuesta = obtenerEstudiantes(); } else if (accion === "guardarAsistencia") { respuesta = guardarAsistencia(parametros.datos); } else if (accion === "guardarReporte") { respuesta = guardarReporte(parametros.datos); } else if (accion === "obtenerEstadisticas") { respuesta = obtenerEstadisticas(); } } catch (error) { respuesta = { exito: false, mensaje: "Error en el servidor: " + error.toString() }; } // Devolvemos la respuesta en formato JSON return ContentService.createTextOutput(JSON.stringify(respuesta)) .setMimeType(ContentService.MimeType.JSON); } // Para evitar problemas de CORS y permitir pruebas por URL function doGet(e) { return ContentService.createTextOutput("La API de asistencia está funcionando correctamente. Usa peticiones POST para interactuar.") .setMimeType(ContentService.MimeType.TEXT); } // --- FUNCIONES AUXILIARES --- function validarUsuario(usuario, password) { let hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Usuarios"); let datos = hoja.getDataRange().getValues(); // Empezamos desde la fila 1 para saltar los encabezados for (let i = 1; i < datos.length; i++) { if (datos[i][0] === usuario && datos[i][1].toString() === password.toString()) { return { exito: true, nombreDocente: datos[i][2] }; } } return { exito: false, mensaje: "Usuario o contraseña incorrectos" }; } function obtenerEstudiantes() { let hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Estudiantes"); // getDisplayValues() trae el texto tal cual se ve en la celda, ignorando el formato de fecha interno let datos = hoja.getDataRange().getDisplayValues(); let estudiantes = []; for (let i = 1; i < datos.length; i++) { estudiantes.push({ nombre: datos[i][0], grado_orig: datos[i][1], grado_new: datos[i][2], acudiente: datos[i][3], tel_acud: datos[i][4] }); } return { exito: true, datos: estudiantes }; } function guardarAsistencia(datosAsistencia) { let hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Asistencia_Log"); // Se espera que datosAsistencia sea un array hoja.appendRow([ datosAsistencia.fecha, datosAsistencia.hora, datosAsistencia.estudiante, datosAsistencia.grado, datosAsistencia.estado, datosAsistencia.docente ]); return { exito: true, mensaje: "Asistencia guardada" }; } function guardarReporte(datosReporte) { let hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Reportes_Conducta"); hoja.appendRow([ datosReporte.fecha, datosReporte.estudiante, datosReporte.acudiente, datosReporte.telefono, datosReporte.falta, datosReporte.observacion, datosReporte.docente ]); return { exito: true, mensaje: "Reporte guardado con éxito" }; } function obtenerEstadisticas() { let hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Asistencia_Log"); let datos = hoja.getDataRange().getDisplayValues(); let logs = []; // Empezamos en 1 para omitir los encabezados for (let i = 1; i < datos.length; i++) { logs.push({ fecha: datos[i][0], hora: datos[i][1], estudiante: datos[i][2], grado: datos[i][3], estado: datos[i][4] }); } return { exito: true, datos: logs }; }
